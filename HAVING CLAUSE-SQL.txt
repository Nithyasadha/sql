1)WAQTD DEPTNO AND NUMBER OF EMPLOYEES IN EACH DEPT IF THERE ARE ATLEAST 2 CLERK IN EACH DEPT
SELECT COUNT(*),DEPTNO FROM EMP WHERE JOB='CLERK' GROUP BY DEPTNO HAVING COUNT(*)>=2;
2)WAQTD DEPTNO AND TOTAL SALARY NEEDED TO PAY ALL EMP IN EACH DEPT IF THERE ARE ATLEAST 4 EMPLOYEES IN EACH DEPT
SELECT SUM(SAL),DEPTNO FROM EMP GROUP BY DEPTNO HAVING COUNT(*)>4;
3)WAQTD NO.OF.EMPLOYEES EARNING SAL MORE THAN 1200 IN EACH JOB AND TOTAL SAL NEEDED TO PAY EMP OF EACH JOB MUST EXCEED 3800
SELECT COUNT(*),JOB FROM EMP WHERE SAL>1200 GROUP BY JOB HAVING SUM(SAL)>3800;
4)WAQTD DEPTNO AND NO.OF.EMPLOYEES WORKING ONLY IF THERE ARE 2 EMP WORKING IN EACH DEPTNO AS MANAGER
SELECT COUNT(*),DEPTNO FROM EMP WHERE JOB='MANAGER' GROUP BY DEPTNO HAVING COUNT(*)=2;
5)WAQTD JOB AND MAX SAL OF EMPLOYEES IN EACH JOB IF THE MAX SAL EXCEEDS 2600
SELECT MAX(SAL),JOB FROM EMP GROUP BY JOB HAVING MAX(SAL)>2600;
6)WAQTD THE SALARIES WHICH ARE REPEATED IN EMPLOYEE TABLE
SELECT SAL FROM EMP GROUP BY SAL HAVING COUNT(*)>1;
7)WAQTD THE HIREDATE WHICH ARE DUPLICATE IN EMP TABLE
SELECT HIREDATE FROM EMP GROUP BY HIREDATE HAVING COUNT(*)>1;
8)WAQTD AVERAGE SALARY OF EACH DEPT IF AVG SAL IS LESS THAN 3000
SELECT AVG(SAL),DEPTNO FROM EMP GROUP BY DEPTNO HAVING AVG(SAL)<3000;
9)WAQTD DEPTNO IF THERE ARE ATLEAST 3 EMPLOYEES IN EACH DEPT WHOSE NAME HAS CHAR 'A' OR 'S'
SELECT DEPTNO FROM EMP WHERE ENAME LIKE '%A' OR ENAME LIKE '%S' GROUP BY DEPTNO HAVING COUNT(*)>=3;
10)WAQTD MIN AND MAX SALARIES OF EACH JOB IF MIN SAL IS MORE THAN 1000 AND MAX SAL IS LESS THAN 5000 
SELECT MIN(SAL),MAX(SAL),DEPTNO FROM EMP GROUP BY DEPTNO HAVING MIN(SAL)>1000 AND MAX(SAL)<5000;
